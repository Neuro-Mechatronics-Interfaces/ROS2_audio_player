[docs]def main(args=None):
    """ Default ROS2 entry point.
    """
    
    # Initialize ROS.
    rclpy.init()
    
    try:
        
        # Create the node.
        node = Node()
        
        # Spin the node.
        try: rclpy.spin(node)
        except KeyboardInterrupt: pass
        finally: node.destroy_node()
        
    finally:
        
        # Shutdown ROS.
        rclpy.shutdown()
        
    return
